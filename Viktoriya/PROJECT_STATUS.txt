🗄️ DATABASE SCHEMA
═══════════════════════════════════════════════════════════════════════════════

employees
├─ id (INT, PK)
├─ employee_code (VARCHAR, UNIQUE)
├─ full_name (VARCHAR)
├─ email (VARCHAR)
├─ role_title (VARCHAR)
├─ department (VARCHAR)
├─ location (VARCHAR)
├─ availability_percent (DECIMAL)
├─ active (BOOLEAN)
└─ timestamps

systems
├─ id (INT, PK)
├─ system_code (VARCHAR, UNIQUE)
├─ name (VARCHAR)
├─ owner_name (VARCHAR)
├─ business_owner (VARCHAR)
├─ technical_owner (VARCHAR)
├─ environment (ENUM: DEV/TEST/UAT/PROD)
├─ sensitivity (ENUM: LOW/MEDIUM/HIGH/CRITICAL)
├─ active (BOOLEAN)
└─ timestamps

services
├─ id (INT, PK)
├─ service_code (VARCHAR, UNIQUE)
├─ name (VARCHAR)
├─ owner_name (VARCHAR)
├─ criticality (ENUM: LOW/MEDIUM/HIGH/CRITICAL)
├─ min_qualified (INT)
├─ preferred_qualified (INT)
├─ active (BOOLEAN)
└─ timestamps

qualifications
├─ id (INT, PK)
├─ employee_id (INT, FK→employees)
├─ system_id (INT, FK→systems)
├─ experience_score (INT)
├─ certification_points (INT)
├─ knowledge_score (INT)
├─ total_score (INT)
├─ qualification_level (ENUM: NONE/BASIC/QUALIFIED/FULLY_CAPABLE/EXPERT)
├─ entry_date (DATE)
├─ review_due_date (DATE)
├─ is_reviewed (BOOLEAN)
├─ UNIQUE(employee_id, system_id)
└─ timestamps

access_reviews
├─ id (INT, PK)
├─ employee_id (INT, FK→employees)
├─ system_id (INT, FK→systems)
├─ access_type (ENUM: NONE/READ/WRITE/ADMIN/OTHER)
├─ requested (BOOLEAN)
├─ approved (BOOLEAN)
├─ reviewed_at (DATETIME)
├─ review_due_date (DATE)
├─ UNIQUE(employee_id, system_id)
└─ timestamps

service_required_systems
├─ id (INT, PK)
├─ service_id (INT, FK→services)
├─ system_id (INT, FK→systems)
├─ order_index (INT)
├─ required_level (ENUM: OPTIONAL/RECOMMENDED/REQUIRED/CRITICAL)
├─ UNIQUE(service_id, system_id)
└─ timestamps

actions
├─ id (INT, PK)
├─ service_id (INT, FK→services, nullable)
├─ employee_id (INT, FK→employees, nullable)
├─ system_id (INT, FK→systems, nullable)
├─ title (VARCHAR)
├─ description (TEXT)
├─ owner_name (VARCHAR)
├─ status (ENUM: OPEN/IN_PROGRESS/BLOCKED/DONE/CANCELLED)
├─ priority (ENUM: LOW/MEDIUM/HIGH/CRITICAL)
├─ due_date (DATE)
├─ completed_at (DATETIME)
└─ timestamps

app_config
├─ id (INT, PK)
├─ config_key (VARCHAR, UNIQUE)
├─ config_value (VARCHAR)
├─ value_type (ENUM: STRING/INTEGER/DECIMAL/BOOLEAN/JSON)
├─ description (TEXT)
└─ timestamps
